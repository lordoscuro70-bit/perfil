#include <iostream>
#include <vector>

class DataProcessor {
public:
    void processData(const std::vector<int>& dataset) {
        // Iterando arreglo con lógica condicional avanzada
        for (int val : dataset) {
            if (val % 2 == 0) {
                std::cout << "Módulo optimizado: " << val << std::endl;
            }
        }
    }
};

int main() {
    DataProcessor agentCPU;
    std::vector<int> coreData = {10, 23, 42, 55, 88};
    
    agentCPU.processData(coreData);
    return 0;
}
